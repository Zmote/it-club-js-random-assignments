Assignment

Write a function that takes a matrix of numbers (in any configuration n x m, n x n, m x n) and randomized it's eyes/values. A number from the source matrix can only be used once in the new matrix.

Example

let matrix = [[1,2],[3,4]];
console.log(randomizeMatrix(matrix)); // possible outputs: --> [[2,3],[1,4]], [[1,3],[2,4]]

Possible solution

    
    function readValues(matrix){
        let values = [];
        for(let i = 0; i < matrix.length;i++){
            for(let j = 0; j < matrix[i].length; j++){
                values.push(matrix[i][j]);
            }
        }
        return values;
    }

    function generateRandomizedMatrix(matrix, values){
        let transformedMatrix = [];
        for(let i = 0; i < matrix.length;i++){
            let column = [];
            for(let j = 0; j < matrix[i].length; j++){
                let randomIndex = Math.floor(Math.random()* values.length);
                column.push(values[randomIndex]);
                values.splice(randomIndex, 1);
            }
            transformedMatrix.push(column);
        }
        return transformedMatrix;
    }

    function randomizeMatrix(matrix){
        let values = readValues(matrix);
        return generateRandomizedMatrix(matrix, values);
    }

    let matrix = [[1,2],[3,4]];
    console.log(randomizeMatrix(matrix));
